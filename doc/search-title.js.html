const video = require("./models/video.js");
exports.searchTitle = async (req, res, next) => {
  const titleSearch = req.query.title;
  const resultOnTitle = await video.fuzzySearch({ "title": titleSearch });
  return resultOnTitle;
};
exports.searchDescription = async (req, res, next) => {
  const descriptionSearch = req.query.description;
  const resultOnDescription = await video.fuzzySearch({ "description": descriptionSearch });
  return resultOnDescription;
};
